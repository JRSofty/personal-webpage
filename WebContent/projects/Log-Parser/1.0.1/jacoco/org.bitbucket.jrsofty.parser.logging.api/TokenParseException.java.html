package org.bitbucket.jrsofty.parser.logging.api;

/**
 * The exception thrown when the token does not match.
 * 
 * @author jrsofty
 *
 */
public class TokenParseException extends Exception {

  private static final long serialVersionUID = 3200062093665668850L;

  public TokenParseException(final String failedToken, final String failedParserType,
      final String regexPattern) {
    super("The token " + failedToken + " could not be parsed by " + failedParserType + " using "
        + regexPattern);
  }

  public TokenParseException(final String failedToken, final String failedParserType,
      final String regexPattern, final Throwable exception) {
    super("The token " + failedToken + " could not be parsed by " + failedParserType + " using "
        + regexPattern, exception);
  }
}
